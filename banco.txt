CREATE TABLE cars_new (
    id_carro SERIAL PRIMARY KEY,
    marca CHAR(15) NOT NULL,
    modelo CHAR(20) NOT NULL,
    tipo CHAR(20) NOT NULL,
    placa CHAR(7) NOT NULL,
    cor CHAR(15) NOT NULL,
    ano CHAR(4) NOT NULL,
    status CHAR(10) NOT NULL,
    priceBuy NUMERIC(10, 2) NOT NULL,
    priceSale NUMERIC(10, 2) NOT NULL,
    priceRent NUMERIC(10, 2) NOT NULL,
    createdAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL
);

TABLE ALUGUEL

CREATE TABLE rented_cars (
    id_aluguel SERIAL PRIMARY KEY,
    id_carro INT NOT NULL REFERENCES cars(id_carro) ON DELETE CASCADE,
    id_usuario INT NOT NULL REFERENCES users(id_usuario) ON DELETE CASCADE,
    data_inicio TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
    data_fim TIMESTAMP NULL,
    status_aluguel CHAR(10) DEFAULT 'ativo' NOT NULL
);
